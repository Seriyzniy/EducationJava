
> [!NOTE] `NavigableMap<K,V>` — это ==интерфейс, который **расширяет== `SortedMap`** и добавляет более гибкие методы для работы с ключами
> 1) Поиск ближайших (floor, ceiling, higher, lower),
> 2) Извлечение и удаление крайних элементов (pollFirst/pollLast).

```java
import java.util.Comparator;
import java.util.Map;
import java.util.SortedMap;
import java.util.NavigableSet;

public interface NavigableMap<K, V> extends SortedMap<K, V> {

	// ✅ Наибольшие ✅
    // Наибольшая пара <K,V> с ключом < key
    Map.Entry<K, V> lowerEntry(K key);
    // Наибольший ключ < key
    K lowerKey(K key);

    // Наибольшая пара с ключом <= key
    Map.Entry<K, V> floorEntry(K key);
    // Наибольший ключ <= key
    K floorKey(K key);
	
	// ✅ Наименьшая ✅
    // Наименьшая пара с ключом >= key
    Map.Entry<K, V> ceilingEntry(K key);
    // Наименьший ключ >= key
    K ceilingKey(K key);

    // Наименьшая пара с ключом > key
    Map.Entry<K, V> higherEntry(K key);
    // Наименьший ключ > key
    K higherKey(K key);

    // ✅ Первая (минимальная) пара
    Map.Entry<K, V> firstEntry();
    // ✅ Последняя (максимальная) пара
    Map.Entry<K, V> lastEntry();

	// ✅Удаление✅
    // Удаляет и возвращает первую пару
    Map.Entry<K, V> pollFirstEntry();
    // Удаляет и возвращает последнюю пару
    Map.Entry<K, V> pollLastEntry();

	// ✅ Порядок следования✅
    // Отображение с обратным порядком
    NavigableMap<K, V> descendingMap();
    // Все ключи в порядке возрастания
    NavigableSet<K> navigableKeySet();
    // Все ключи в порядке убывания
    NavigableSet<K> descendingKeySet();

	// ✅ Части карты ✅
    // Подкарта от fromKey до toKey (границы включительно/исключительно)
    NavigableMap<K, V> subMap(K fromKey, boolean fromInclusive,
                              K toKey, boolean toInclusive);
                              
    // Подкарта от начала до toKey (вкл/искл)
    NavigableMap<K, V> headMap(K toKey, boolean inclusive);
    
    // Подкарта от fromKey до конца (вкл/искл)
    NavigableMap<K, V> tailMap(K fromKey, boolean inclusive);
}
```
