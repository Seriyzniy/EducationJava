```sql
CREATE TABLE Articles(
	a_id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
	a_title varchar(30), 
	a_description varchar(30), 
	a_content TEXT, 
	a_updatedate DATE, 
	a_updatetime TIME,
	a_author_id int REFERENCES Persons (p_id)
)
```

```sql
CREATE TABLE Persons(
	p_id INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
	p_name VARCHAR(25) NOT NULL UNIQUE,
	p_pass VARCHAR(255) NOT NULL,
	p_role VARCHAR(25) NOT NULL
)
```
