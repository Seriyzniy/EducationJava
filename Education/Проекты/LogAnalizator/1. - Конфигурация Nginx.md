## docker-compose
```yaml
version: "3.5"  
  
services:  
  nginx-services:  
    image: nginx:latest  
    container_name: nginx-container  
    ports:  
      - 8081:80  
  
    volumes:  
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf  
      - ./nginx/pages/errors:/usr/share/nginx/html/errors  
      - ./nginx/pages/success:/usr/share/nginx/html/success  
      - ./nginx/logs:/var/log/nginx
```

## nginx.conf
```
user nginx;  
worker_processes auto;  
  
events {  
    worker_connections 1024;  
}  
  
http {  
    include       /etc/nginx/mime.types;  
    default_type  text/html;  
  
    sendfile on;  
    keepalive_timeout 65;  
  
    # ---------- ЛОГИРОВАНИЕ ----------  
    map $status $is_success {  
        default 0;  
        ~^[1-3] 1;  
    }  
  
    map $status $is_error {  
        default 0;  
        ~^[45] 1;  
    }  
  
    access_log /var/log/nginx/access.log combined if=$is_success;  
    access_log /var/log/nginx/error.log  combined if=$is_error;  
  
    server {  
        listen 80;  
        server_name localhost;  
  
        # ---------- КАСТОМНЫЕ СТРАНИЦЫ ----------        
        error_page 400 /errors/400.html;  
        error_page 403 /errors/403.html;  
        error_page 404 /errors/404.html;  
        error_page 500 /errors/500.html;  
        error_page 502 /errors/502.html;  
        error_page 503 /errors/503.html;  
        error_page 504 /errors/504.html;  
  
        location /errors/ {  
            root /usr/share/nginx/html;  
            internal;  
        }  
  
        # ---------- УСПЕШНЫЕ ----------        
	    location = /200 {  
            root /usr/share/nginx/html;  
            try_files /success/200.html =200;  
        }  
  
        # ---------- ОШИБКИ ----------        
        location = /400 { return 400; }        
        location = /403 { return 403; }        
        location = /404 { return 404; }        
        location = /500 { return 500; }        
        location = /502 { return 502; }        
        location = /503 { return 503; }        
        location = /504 { return 504; }  
        
        location / {  
            return 404;  
        }  
    }  
}
```
