# 1. Создание таблиц
```sql
CREATE TABLE Clients(
	client_id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
	c_name varchar(25) NOT NULL,
	c_age int CHECK(c_age < 100 AND c_age > 0)
);
INSERT INTO Clients(c_name, c_age) VALUES ('Tom', 24)
```

```sql
CREATE TABLE Items(
	item_id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
	fk_client_id int REFERENCES Clients(client_id) ON DELETE SET NULL,
	item_name varchar(25) NOT NULL
);
INSERT INTO Items(fk_client_id, item_name) VALUES (1, 'Phone');
INSERT INTO Items(fk_client_id, item_name) VALUES (1, 'Notebook');
INSERT INTO Items(fk_client_id, item_name) VALUES (1, 'Pen');
```

# 2. Создание классов сущностей
```java
@Entity
@Table(name = "Clients")
public class Client {
	@Id
	@Column(name = "client_id")
	@GeneratedValue(strategy = GenerationType.IDENTITY)
	private int id;
	
	@Column(name = "c_name")
	private String name;
	
	@Column(name = "c_age")
	private int age;
	
	@OneToMany(mappedBy = "client")
	private List<Item> c_items;
	
	public Client() {};
	public Client(String name, int age) {
		this.name = name;
		this.age = age;
	}
	
	public int getAge() {
		return age;
	}
	public int getId() {
		return id;
	}
	public String getName() {
		return name;
	}
	public List<Item> getC_items() {
		return c_items;
	}
	public void setAge(int age) {
		this.age = age;
	}
	public void setName(String name) {
		this.name = name;
	}
	public void setC_items(List<Item> c_items) {
		this.c_items = c_items;
	}
	public String toString() {
		return this.id + ", " +this.name + ", " + this.age;
	}
}
```

```java
@Entity
@Table(name = "Items")
public class Item {
	@Id
	@Column(name = "item_id")
	@GeneratedValue(strategy = GenerationType.IDENTITY)
	private int item_id;
	
	@Column(name = "item_name")
	private String name;
	
	@ManyToOne
	@JoinColumn(name = "fk_client_id", referencedColumnName = "client_id")
	private Client client;
	
	public Item() {};
	public Item(String name) {
		this.name = name;
	}
	
	public int getId() {
		return item_id;
	}
	public String getName() {
		return name;
	}
	public Client getClient() {
		return client;
	}
	public void setName(String name) {
		this.name = name;
	}
	public void setClient(Client client) {
		this.client = client;
	}
	public String toString() {
		return this.item_id + ", " + this.name;
	}
}
```

# 3. Просмотр загрузки ПО УМОЛЧАНИЮ со СТОРОНЫ ==КЛИЕНТА==

```java
public class App {
	public static void main(String[] args) {
		Configuration myConfigHiber = new Configuration();
		myConfigHiber.addAnnotatedClass(Client.class);
		myConfigHiber.addAnnotatedClass(Item.class);
		
		SessionFactory sessionFactory = myConfigHiber.buildSessionFactory();
		Session session = sessionFactory.getCurrentSession();
		try {
			session.beginTransaction();
			
			Client client = session.get(Client.class, 1);
			System.out.println("Получили клеинта!");
			//Получим связанные товары
			System.out.println(client.getC_items());
			
			session.getTransaction().commit();
		}finally{
			sessionFactory.close();
		}
	}
}
```

Из вывода видно, что сначала выполняется запрос к БД для получения ТОЛЬКО КЛИЕНТА. После получения клиента, мы в коде на ПРЯМУЮ ОБРАЩАЕМСЯ к списку товаров, поэтому генерируется еще один SQL-код.

В данном примере использовалась ленивая загрузка. Мы со стороны клиента, которая относится как "1 ко М" получили список товаров

# 5. Использование НЕ ЛЕНИВОЙ загрузки на стороне КЛИЕНТА

Для этого добавим параметр `fetch`
```java
class Client{
	. . .
	@OneToMany(mappedBy = "client", fetch = FetchType.EAGER)
	private List<Item> c_items;
	. . .
}
```

```java
class App{
	try {
		session.beginTransaction();
		
		Client client = session.get(Client.class,1);
		System.out.println("Получили товар");
		System.out.println(client.getC_items());
	
		session.getTransaction().commit();
	}finally{
}
```
