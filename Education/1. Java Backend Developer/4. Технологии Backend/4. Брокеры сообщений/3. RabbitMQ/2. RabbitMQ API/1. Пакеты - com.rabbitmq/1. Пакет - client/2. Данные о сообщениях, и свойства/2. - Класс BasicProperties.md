
> [!NOTE] Класс `BasicProperties`
> **BasicProperties** - это класс, представляющий свойства сообщения AMQP. 
> 
> ==Он инкапсулирует все МЕТАДАННЫЕ сообщения==, такие как **заголовки**, **тип контента**, **идентификатор** сообщения, **timestamp** и другие атрибуты. Это один из самых важных классов для настройки поведения сообщений в RabbitMQ.

**Зачем нужен:** Для задания метаданных сообщения при публикации и получения этих метаданных при потреблении.
> Свойства управляют такими аспектами как персистентность сообщений, приоритет, срок жизни (TTL), маршрутизация и многое другое.

**Класс является вложенным (nested) классом:** `AMQP.BasicProperties`  
**Имеет вложенный Builder класс:** `AMQP.BasicProperties.Builder` для удобного создания объектов.

# Конструктор
```java
/**
 * Constructor that takes all possible values.
 *
 * @param contentType content type
 * @param contentEncoding content encoding
 * @param headers message header field table
 * @param deliveryMode delivery mode (1: non-persistent, 2: persistent)
 * @param priority message priority, 0 to 9
 * @param correlationId application correlation identifier
 * @param replyTo address to reply to
 * @param expiration message expiration specification
 * @param messageId application message identifier
 * @param timestamp message timestamp
 * @param type message type name
 * @param userId creating user id
 * @param appId creating application id
 * @param clusterId intra-cluster routing identifier (deprecated)
 */
public BasicProperties(String contentType,
                       String contentEncoding,
                       Map<String,Object> headers,
                       Integer deliveryMode,
                       Integer priority,
                       String correlationId,
                       String replyTo,
                       String expiration,
                       String messageId,
                       Date timestamp,
                       String type,
                       String userId,
                       String appId,
                       String clusterId) {
    this.contentType = contentType;
    this.contentEncoding = contentEncoding;
    this.headers = headers;
    this.deliveryMode = deliveryMode;
    this.priority = priority;
    this.correlationId = correlationId;
    this.replyTo = replyTo;
    this.expiration = expiration;
    this.messageId = messageId;
    this.timestamp = timestamp;
    this.type = type;
    this.userId = userId;
    this.appId = appId;
    this.clusterId = clusterId;
}
// Создает новый BasicProperties со всеми возможными значениями.
```
💬**Комментарий:** Конструктор принимает все возможные параметры, многие из которых могут быть `null`. 
На практике для создания объектов `BasicProperties` ==почти всегда используется вложенный== `Builder` класс, который предоставляет fluid API для удобного конструирования.
## **Поля класса** (все приватные и final в иммутабельной версии):

- `contentType`, `contentEncoding` - тип и кодировка контента
    
- `headers` - дополнительные заголовки (Map<String, Object>)
    
- `deliveryMode` - режим доставки (1 - non-persistent, 2 - persistent)
    
- `priority` - приоритет сообщения (0-9)
    
- `correlationId`, `replyTo` - для RPC паттерна
    
- `expiration` - TTL в миллисекундах (строка)
    
- `messageId`, `timestamp`, `type`, `userId`, `appId`, `clusterId` - различные идентификаторы

## Builder (Строитель)

### Класс Builder
```java
/**
 * Builder for {@link BasicProperties}.
 */
public static class Builder {
    // Поля Builder соответствуют полям BasicProperties
    
    /**
     * Start a new builder with no properties set.
     */
    public Builder() { }
    // Создает новый строитель без установленных свойств.

    /**
     * Start a new builder with the properties from an existing
     * BasicProperties instance.
     * @param properties the properties to copy
     */
    public Builder(BasicProperties properties) { }
    // Создает новый строитель, копируя свойства из существующего экземпляра BasicProperties.

    // Все методы Builder возвращают this для chaining

    public Builder contentType(String contentType) { return this; }
    // Устанавливает тип контента.

    public Builder contentEncoding(String contentEncoding) { return this; }
    // Устанавливает кодировку контента.

    public Builder headers(Map<String,Object> headers) { return this; }
    // Устанавливает заголовки сообщения.

    public Builder deliveryMode(Integer deliveryMode) { return this; }
    // Устанавливает режим доставки (1: non-persistent, 2: persistent).

    public Builder priority(Integer priority) { return this; }
    // Устанавливает приоритет сообщения (0-9).

    public Builder correlationId(String correlationId) { return this; }
    // Устанавливает идентификатор корреляции.

    public Builder replyTo(String replyTo) { return this; }
    // Устанавливает адрес для ответа.

    public Builder expiration(String expiration) { return this; }
    // Устанавливает срок истечения (TTL) в миллисекундах.

    public Builder messageId(String messageId) { return this; }
    // Устанавливает идентификатор сообщения.

    public Builder timestamp(Date timestamp) { return this; }
    // Устанавливает временную метку.

    public Builder type(String type) { return this; }
    // Устанавливает тип сообщения.

    public Builder userId(String userId) { return this; }
    // Устанавливает идентификатор пользователя.

    public Builder appId(String appId) { return this; }
    // Устанавливает идентификатор приложения.

     * Build a new instance of BasicProperties with the currently-set
     * configuration.
     * @return a new BasicProperties instance
     */
    public BasicProperties build() { return null; }
    // Создает новый экземпляр BasicProperties с текущей конфигурацией.
}
```
### Создание и методы Builder
```java
/**
 * Retrieve a builder that can be used to create a new instance of BasicProperties.
 * @return a new builder
 */
public static Builder builder() {
    return new Builder();
}
// Возвращает новый строитель (builder), который может быть использован для создания нового экземпляра BasicProperties.

/**
 * Retrieve a builder, pre-initialised with the contents of "this".
 * @return a new builder
 */
public Builder builder() {
    return new Builder(this);
}
// Возвращает новый строитель, предварительно инициализированный содержимым "this".
```

# 💡 Особенности и подводные камни 💡

1️⃣ **Иммутабельность (начиная с определенной версии):** В современных версиях RabbitMQ Java Client класс `BasicProperties` является иммутабельным. Это означает, что после создания объект нельзя изменить:

```java
// ❌ НЕ РАБОТАЕТ - класс иммутабельный
properties.setContentType("application/json");


// ✅ ПРАВИЛЬНО - создаем новый объект через Builder
BasicProperties newProps = properties.builder()
    .contentType("application/json")
    .build();
```

2️⃣ **Null значения:** Все поля могут быть `null`. При получении значений всегда проверяйте на `null`:

```java
// ❌ ОПАСНО - может вызвать NPE
String contentType = properties.getContentType().toLowerCase();


// ✅ БЕЗОПАСНО
if (properties.getContentType() != null) {
    String contentType = properties.getContentType().toLowerCase();
}
```

3️⃣ 🔥**Delivery Mode критически важен для persistence:**🔥
- `1` - non-persistent (сообщение теряется при перезагрузке брокера)
    
- `2` - persistent (сообщение сохраняется на диск)

```java
// Персистентное сообщение
BasicProperties props = new BasicProperties.Builder()
    .deliveryMode(2)  // ← ВАЖНО!
    .build();
```

4️⃣ **Expiration - строка, а не число:** Поле `expiration` хранится как строка, содержащая количество миллисекунд:
```java
// Устанавливаем TTL 5 минут (300000 мс)
BasicProperties props = new BasicProperties.Builder()
    .expiration("300000")  // ← СТРОКА!
    .build();
```

5️⃣ **Headers - могут содержать сложные типы:** Заголовки поддерживают различные типы данных:
```java
Map<String, Object> headers = new HashMap<>();

headers.put("string-header", "value");
headers.put("int-header", 42);
headers.put("boolean-header", true);
headers.put("list-header", Arrays.asList("a", "b", "c"));
headers.put("map-header", Collections.singletonMap("key", "value"));
```

6️⃣ **Priority работает только с приоритетными очередями:** Приоритет сообщения учитывается только если очередь объявлена с поддержкой приоритетов:
```java
Map<String, Object> args = new HashMap<>();
args.put("x-max-priority", 10);  // Максимальный приоритет 10
channel.queueDeclare("priority-queue", true, false, false, args);

// Сообщение с приоритетом
BasicProperties props = new BasicProperties.Builder()
    .priority(5)  // Будет учтен только в priority-queue
    .build();
```

7️⃣ **Timestamp - Date объект, но передается как long:** Хотя поле имеет тип `Date`, при сериализации в AMQP передается timestamp в секундах:
```java
// RabbitMQ преобразует Date.getTime() / 1000
Date now = new Date();  // Текущее время с миллисекундами
// В AMQP будет передано now.getTime() / 1000 (секунды)
```

8️⃣ **ContentType vs ContentEncoding:** Частая путаница:
- `contentType` - тип данных (application/json, text/plain, etc.)
    
- `contentEncoding` - кодировка (utf-8, gzip, etc.)

# 💎 Best Practices 💎

✅ Всегда ==используйте Builder для создания BasicProperties==:
```java
// Простой случай
BasicProperties props = new BasicProperties.Builder()
    .contentType("application/json")
    .deliveryMode(2)
    .timestamp(new Date())
    .build();

// Сложный случай с заголовками
Map<String, Object> headers = new HashMap<>();
headers.put("trace-id", UUID.randomUUID().toString());
headers.put("retry-count", 0);

BasicProperties props = new BasicProperties.Builder()
    .contentType("application/json")
    .contentEncoding("utf-8")
    .deliveryMode(2)
    .priority(5)
    
    .headers(headers)
    
    .messageId(UUID.randomUUID().toString())
    .timestamp(new Date())
    .appId("order-service")
    .build();
```

✅ **Создавайте фабричные методы для часто используемых конфигураций:**
```java
public class MessagePropertiesFactory {
    public static BasicProperties persistentJsonMessage() {
        return new BasicProperties.Builder()
            .contentType("application/json")
            .deliveryMode(2)
            .build();
    }
    
    public static BasicProperties rpcRequest(String correlationId) {
        return new BasicProperties.Builder()
            .contentType("application/json")
            .deliveryMode(2)
            .correlationId(correlationId)
            .replyTo("rpc-replies")
            .timestamp(new Date())
            .build();
    }
    
    public static BasicProperties withTtl(String ttlMillis) {
        return new BasicProperties.Builder()
            .deliveryMode(2)
            .expiration(ttlMillis)
            .build();
    }
}
```

✅ **Проверяйте deliveryMode для критичных сообщений:**
```java
public void ensurePersistent(AMQP.BasicProperties properties) {
    if (properties.getDeliveryMode() == null || properties.getDeliveryMode() != 2) {
        throw new IllegalArgumentException("Message must be persistent (deliveryMode=2)");
    }
}
```

✅ **Используйте заголовки для сквозной трассировки:**
```java
public BasicProperties withTracingHeaders(String traceId, String spanId) {
    Map<String, Object> headers = new HashMap<>();
    if (this.getHeaders() != null) {
        headers.putAll(this.getHeaders());
    }
    headers.put("X-Trace-Id", traceId);
    headers.put("X-Span-Id", spanId);
    
    return this.builder()
        .headers(headers)
        .build();
}
```

✅ **Обрабатывайте expiration безопасно:**
```java
public Long getExpirationMillis(AMQP.BasicProperties properties) {
    if (properties.getExpiration() == null) {
        return null;
    }
    try {
        return Long.parseLong(properties.getExpiration());
    } catch (NumberFormatException e) {
        logger.warn("Invalid expiration format: {}", properties.getExpiration());
        return null;
    }
}
```

✅ **Используйте correlationId для RPC и трассировки:**
```java
// При отправке RPC запроса
String correlationId = UUID.randomUUID().toString();
BasicProperties props = new BasicProperties.Builder()
    .correlationId(correlationId)
    .replyTo(replyQueueName)
    .build();

// Сохраняем correlationId для сопоставления ответа
pendingRequests.put(correlationId, request);

// При получении ответа
String receivedCorrelationId = properties.getCorrelationId();
if (receivedCorrelationId != null) {
    Request originalRequest = pendingRequests.remove(receivedCorrelationId);
    // Обработка ответа
}
```

✅ **Валидируйте заголовки при получении сообщений:**
```java
public void validateHeaders(AMQP.BasicProperties properties) {
    if (properties.getHeaders() != null) {
        properties.getHeaders().forEach((key, value) -> {
            if (value instanceof LongString) {
                // LongString - специальный тип RabbitMQ для больших строк
                logger.debug("Header {} is LongString", key);
            }
            // Дополнительная валидация типов
        });
    }
}
```

✅ **Кэшируйте часто используемые BasicProperties:**
```java
public class PropertiesCache {
    private static final BasicProperties PERSISTENT_TEXT = 
        new BasicProperties.Builder()
            .contentType("text/plain")
            .deliveryMode(2)
            .build();

    public static BasicProperties persistentText() {
        return PERSISTENT_TEXT;
    }
    
    public static BasicProperties persistentJson() {
        return PERSISTENT_JSON;
    }
}
```

# Пример использования
```java
// Пример комплексного использования BasicProperties
import com.rabbitmq.client.AMQP;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;
import java.util.UUID;

public class MessagePublisher {
    
    public void publishMessage() throws Exception {
        // Создание заголовков
        Map<String, Object> headers = new HashMap<>();
        headers.put("trace-id", UUID.randomUUID().toString());
        headers.put("retry-count", 0);
        
        // ПРАВИЛЬНО: Использование AMQP.BasicProperties
        AMQP.BasicProperties properties = new AMQP.BasicProperties.Builder()
                .contentType("application/json")
                .contentEncoding("utf-8")
                .deliveryMode(2)  // Persistent (2), Non-persistent (1)
                .priority(5)      // 0-9, где 9 - наивысший
                .correlationId(UUID.randomUUID().toString())
                .replyTo("reply-queue")
                .expiration("300000")  // TTL 5 минут (в миллисекундах как String)
                .messageId(UUID.randomUUID().toString())
                .timestamp(new Date())
                .type("order.created")
                .userId("system")
                .appId("order-service")
                .headers(headers)
                .build();
        
        // Использование properties
        channel.basicPublish(
            "exchange-name",
            "routing.key",
            true,  // mandatory
            properties,
            "message body".getBytes()
        );
    }
    
    // Альтернативный вариант со статическим импортом
    public void publishWithStaticImport() throws Exception {
        // Статический импорт
        import static com.rabbitmq.client.AMQP.BasicProperties;
        
        BasicProperties props = BasicProperties.builder()
                .contentType("text/plain")
                .deliveryMode(2)
                .build();
    }
    
    // Получение свойств (работает с интерфейсом)
    public void processMessage(com.rabbitmq.client.BasicProperties properties) {
        // Можно использовать интерфейс для чтения
        String contentType = properties.getContentType();
        Integer deliveryMode = properties.getDeliveryMode();
        
        // Но НЕЛЬЗЯ создавать или модифицировать через интерфейс
        // properties.setContentType("new-type"); // ❌ Не существует!
    }
}
```
