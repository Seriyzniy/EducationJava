# 1. Подключение зависимостей
Как и в [[0. - Подключение jar файлов к проекту]] нам нужны следующие зависимости
- Kafka CLients
- SLF4J (Prodvider и API)

Найдем эти зависимости в Maven Repository, и добавим в pom.xml
```xml
<dependency>  
    <groupId>org.apache.kafka</groupId>  
    <artifactId>kafka-clients</artifactId>  
    <version>3.9.1</version>  
</dependency>  
  
<dependency>  
    <groupId>org.slf4j</groupId>  
    <artifactId>slf4j-api</artifactId>  
    <version>2.0.17</version>  
</dependency>  
<dependency>  
    <groupId>org.slf4j</groupId>  
    <artifactId>slf4j-simple</artifactId>  
    <version>2.0.17</version>  
</dependency>
```

Для сериализации и десериализации данных БУДЕМ ИСПОЛЬЗОВАТЬ Jackson
```xml
<dependency>  
    <groupId>com.fasterxml.jackson.core</groupId>  
    <artifactId>jackson-databind</artifactId>  
    <version>2.20.1</version>  
</dependency>  
<dependency>  
    <groupId>com.fasterxml.jackson.core</groupId>  
    <artifactId>jackson-core</artifactId>  
    <version>2.20.1</version>  
</dependency>

<!-- Для сериализации временных типов - Instant и прочие -->  
<dependency>  
    <groupId>com.fasterxml.jackson.datatype</groupId>  
    <artifactId>jackson-datatype-jsr310</artifactId>  
    <version>2.20.1</version> <!-- или актуальная версия -->  
</dependency>
```

# 2. Настройка кластера
В качестве кластера Kafka будем использовать кластер, созданный при написании проекта Spring+Kafka

Вот его конфигурация docker-compose
```yaml
version: "3.5"
services:
  kafka-bc-service-1:
    image: apache/kafka:latest
    container_name: kafka-bc-container-1
    ports:
      - "29092:9092"
    environment:
      KAFKA_NODE_ID: 1
      KAFKA_PROCESS_ROLES: broker,controller

      KAFKA_LISTENERS:
        MY_HOST://:9092,
        MY_INTERNAL_HOST://:19092, 
        CONTROLLER://:9093
      KAFKA_ADVERTISED_LISTENERS:
        MY_HOST://localhost:29092,
        MY_INTERNAL_HOST://kafka-bc-container-1:19092
      KAFKA_LISTENER_SECURITY_PROTOCOL_MAP:
        MY_HOST:PLAINTEXT,
        MY_INTERNAL_HOST:PLAINTEXT,
        CONTROLLER:PLAINTEXT

      KAFKA_INTER_BROKER_LISTENER_NAME: MY_INTERNAL_HOST
      KAFKA_CONTROLLER_LISTENER_NAMES: CONTROLLER

      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka-bc-container-1:9093

      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_MIN_ISR: 1


  kafka-b-service-2:
    image: apache/kafka:latest
    container_name: kafka-b-container-2
    ports:
      - "39092:9092"
    environment:
      KAFKA_NODE_ID: 2
      KAFKA_PROCESS_ROLES: broker

      KAFKA_LISTENERS:
        MY_HOST://:9092,
        MY_INTERNAL_HOST://:19092
      KAFKA_ADVERTISED_LISTENERS:
        MY_HOST://localhost:39092,
        MY_INTERNAL_HOST://kafka-b-container-2:19092
      KAFKA_LISTENER_SECURITY_PROTOCOL_MAP:
        MY_HOST:PLAINTEXT,
        MY_INTERNAL_HOST:PLAINTEXT,
        CONTROLLER:PLAINTEXT

      KAFKA_INTER_BROKER_LISTENER_NAME: MY_INTERNAL_HOST
      KAFKA_CONTROLLER_LISTENER_NAMES: CONTROLLER

      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka-bc-container-1:9093

      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_MIN_ISR: 1


  kafka-b-service3:
    image: apache/kafka:latest
    container_name: kafka-b-container-3
    ports:
      - "49092:9092"
    environment:
      KAFKA_NODE_ID: 3
      KAFKA_PROCESS_ROLES: broker

      KAFKA_LISTENERS:
        MY_HOST://:9092,
        MY_INTERNAL_HOST://:19092
      KAFKA_LISTENER_SECURITY_PROTOCOL_MAP:
        MY_HOST:PLAINTEXT,
        MY_INTERNAL_HOST:PLAINTEXT,
        CONTROLLER:PLAINTEXT

        MY_HOST://localhost:49092,
        MY_INTERNAL_HOST://kafka-b-container-3:19092

      KAFKA_INTER_BROKER_LISTENER_NAME: MY_INTERNAL_HOST
      KAFKA_CONTROLLER_LISTENER_NAMES: CONTROLLER

      KAFKA_CONTROLLER_QUORUM_VOTERS: 1@kafka-bc-container-1:9093

      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_MIN_ISR: 1
    depends_on:
      - kafka-bc-service-1
      - kafka-b-service-2

  kafka-ui:
    image: provectuslabs/kafka-ui:latest
    container_name: kafka-ui-for-spring-example
    ports:
      - "8080:8080"
    environment:
      KAFKA_CLUSTERS_0_NAME: "local"
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: "kafka-bc-container-1:19092"
```
