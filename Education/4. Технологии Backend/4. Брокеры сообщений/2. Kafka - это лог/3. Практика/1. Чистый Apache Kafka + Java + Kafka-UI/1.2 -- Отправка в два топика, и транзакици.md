```java
package producer;

import org.apache.kafka.clients.producer.*;
import org.apache.kafka.common.serialization.StringSerializer;

import java.util.Properties;

public class MyProducer {
    private final KafkaProducer<String, String> producer;

    public MyProducer() {
        Properties props = new Properties();
        props.put(
            ProducerConfig.BOOTSTRAP_SERVERS_CONFIG,
            "localhost:9092");
        props.put(
            ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG,
            StringSerializer.class.getName());
        props.put(
            ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG,
            StringSerializer.class.getName());

        producer = new KafkaProducer<>(props);
    }

    public void produceMessaging(){
        try(producer){
            ProducerRecord<String, String> record1 =
                new ProducerRecord<>(
                    "topic-1",
                    "Message to Topic-1");

            ProducerRecord<String, String> record2 =
                new ProducerRecord<>(
                    "topic-2",
                    "Message to Topic-2");

            producer.send(record1);
            producer.send(record2);
            System.out.println("[INFO] Message sent");

        } catch (Exception e){
            System.out.println("Exception on produce message");
        }
    }

    public static void main(String[] args) {
        MyProducer myProducer = new MyProducer();
        myProducer.produceMessaging();
    }
}
```

```java
import java.util.Collections;

import java.util.Properties;

        Properties props = new Properties();  
        props.put(

            "localhost:9092"  
        );  
        props.put(  
            ConsumerConfig.KEY_DESERIALIZER_CLASS_CONFIG,

        props.put(  
            ConsumerConfig.VALUE_DESERIALIZER_CLASS_CONFIG,  
            StringDeserializer.class  
        );  
        props.put(ConsumerConfig.GROUP_ID_CONFIG, "group-2");  
        props.put(
	        ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, 
	        "earliest"
	    );  
  
        consumer = new KafkaConsumer<>(props);  
    }  
  
    public void consumeMessages() {

            consumer.subscribe(List.of("topic-1", "topic-2"));  
            System.out.println(
	            "[INFO] Subscribed to topic:"+

            while(true){  
                ConsumerRecords<String, String> records =  
                        consumer.poll(Duration.ofMillis(1000));  
                System.out.println(
	                "[INFO] Polled records: "

	            ){

                        System.out.println(
	                        "[INFO] Message: " +
	                        record.value());  
                    }  
  
                    System.out.println();  
                }  
            }  
  
        } catch (Exception e){  
            System.out.println("Exception on consume message");  
        }  
    }  
  
    public static void main(String[] args) {  
        MyConsumer myConsumer = new MyConsumer();  
        myConsumer.consumeMessages();  
    }  
}
```

# 📕2. Симуляция ошибки, и подключение транзакций

## 1. Producer
Добавим в конфигурацию Producer параметры, необходимые для Kafka Transaction
(см инфу про транзакции в Kafka [[2. - Kafka Transaction]])

А именно
- TRANSACTIONAL_ID_CONFIG
- ENABLE_IDEMPOTENCE_CONFIG

```java
package producer;  
  
import org.apache.kafka.clients.producer.*;  
import org.apache.kafka.common.serialization.StringSerializer;  
  
import java.util.Properties;  
  
public class MyProducer {  
    private final KafkaProducer<String, String> producer;  
  
    public MyProducer() {  
        Properties props = new Properties();  
        props.put(  
            ProducerConfig.BOOTSTRAP_SERVERS_CONFIG,  
            "localhost:9092");  
        props.put(  
            ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG,  
            StringSerializer.class.getName());  
        props.put(  
            ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG,  
            StringSerializer.class.getName());  
  
        // 1️⃣Конфигурация транзакции  
        props.put(  
            ProducerConfig.TRANSACTIONAL_ID_CONFIG,  
            "producer-transactional-id");  
        props.put(  
            ProducerConfig.ENABLE_IDEMPOTENCE_CONFIG,  
            "true");  
  
        producer = new KafkaProducer<>(props);  
    }  
  
    public void produceMessaging(){  
        try {  
            producer.initTransactions();  
            producer.beginTransaction();  
  
            ProducerRecord<String, String> record1 =  
                new ProducerRecord<>(  
                    "topic-1",  
                    "Message to Topic-1");  
  
            ProducerRecord<String, String> record2 =  
                new ProducerRecord<>(  
                    "topic-2",  
                    "Message to Topic-2");  
  
  
            producer.send(record1, (metadata, exception) -> {
                if(exception == null) {
                    System.out.println(
	                    "[INFO] Message successfully sent to topic-1");
                } else {
                    System.out.println(
	                    "[ERROR] Error while sending message to topic-1");
                }
            });
            producer.send(record2, (metadata, exception) -> {
                if(exception == null) {
                    System.out.println(
	                    "[INFO] Message successfully sent to topic-2");
                } else {
                    System.out.println(
	                    "[ERROR] Error while sending message to topic-2");
                }
            }); 
            System.out.println("[INFO] Message sent");  
			
			// 2️⃣Симуляция ошибки
			// throw наярпямую нелзя, 
			// иначе не компилируется программу
            int x = 1/0;  
  
            producer.commitTransaction();  
  
        } catch (Exception e){  
            System.out.println("Exception on produce message");  
            producer.abortTransaction();  
        } finally {  
            producer.close();  
        }  
    }  
  
    public static void main(String[] args) {  
        MyProducer myProducer = new MyProducer();  
        myProducer.produceMessaging();  
    }  
}
```

![[Pasted image 20251116144539.png]]
Как видим, сообщение в топик 1 успешно отправилось, а в топик 2 не отправилось. 
Скорее всего, потому что первым отправилось сообщение 1, а потом сообщение 2. Поэтому процесс подтверждения для первого сообщения начался раньше чем ACK у второго.
В результате, к моменту когда транзакция начала откатываться, первое сообщение успело сохраниться на брокере, а второе нет, из-за чего транзакция откатила отправку второго сообщения

Однако, Consumer все же получает сообщение из топика 1, так как оно успело сораниться
![[Pasted image 20251116144937.png]]

Это происходит, потому что Consumer по умолчанию работает в режиме `read_uncommited`

Чтобы COnsumer считывал только сообщения, которые были завершены в транзакции добавим в конфигурацию Consumer параметры
- ==ISOLATION_LEVEL_CONFIG== = "read_committed"

```java
public MyConsumer() {
```
